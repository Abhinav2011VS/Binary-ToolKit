function isJavaScript(input) { try { // Try to parse the input as JavaScript new Function(input); return true; } catch (error) { // If parsing fails, it's not JavaScript return false; } } function convertToBinary(input) { if (isJavaScript(input)) { // Convert JavaScript to Binary const binary = input.split('').map(char => char.charCodeAt(0).toString(2)).join(' '); return binary; } else { return 'Please enter valid JavaScript code.'; } } function convertToJavaScript(input) { if (/^[01\s]+$/.test(input)) { // Convert Binary to JavaScript const binary = input.split(' ').map(binary => String.fromCharCode(parseInt(binary, 2))).join(''); if (isJavaScript(binary)) { return binary; } else { return 'The converted binary does not represent valid JavaScript code.'; } } else { return 'Please enter valid binary code.'; } }
